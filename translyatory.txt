PROGRAM Main
VAR
    count INT = 1;
    result INT;
END_VAR

WHILE count = 25 DO
    IF count MOD 2 = 0 THEN
        result = count  2;
        PRINT('Четное число ', count, ', умноженное на 2 равно ', result);
    ELSE
        result = count  3;
        PRINT('Нечетное число ', count, ', умноженное на 3 равно ', result);
    END_IF

    count = count + 1;
    PRINT('Итерация завершена ', count);
END_WHILE

PRINT('Завершение программы');

END_PROGRAM
